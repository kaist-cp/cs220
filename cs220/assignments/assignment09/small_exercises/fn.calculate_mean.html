Returns mean for each class.

§Example

use cs220::assignments::assignment09::small_exercises::*;

assert_eq!(
    calculate_mean(
        [
            ("CS100".to_string(), 60),
            ("CS200".to_string(), 60),
            ("CS200".to_string(), 80),
            ("CS300".to_string(), 100),
        ]
        .into_iter()
    ),
    [
        ("CS100".to_string(), 60.0),
        ("CS200".to_string(), 70.0),
        ("CS300".to_string(), 100.0)
    ]
    .into_iter()
    .collect()
);
